# YTChat TTS — dependencias de Python
#
# Instalar con:
#   pip install -r requirements.txt
#
# Todo funciona con Python 3.11+ de 64 bits en Windows 10/11.

# Captura del chat en vivo de YouTube (InnerTube API)
pytchat>=0.5.5

# Acceso a SAPI5 y COM (Windows)
pywin32>=305

# Interfaz gráfica accesible
wxPython>=4.2.2

# Puente con los lectores de pantalla NVDA y JAWS.
# Opcional: si no está instalado o no hay lector activo, el programa
# funciona igualmente pero sin anuncios hablados complementarios.
accessible_output2>=0.17

# ── Funciones online (OPCIONAL): YouTube Data API v3 ──────────────────────────
# Necesarias solo para leer comentarios de vídeos, moderar el chat en vivo,
# enviar mensajes al directo y publicar/responder comentarios. Si no se
# instalan, esas funciones quedan desactivadas y el resto de la app funciona
# igual. Ver docs/CONFIGURACION_API.md para activarlas.
google-api-python-client>=2.100
google-auth-oauthlib>=1.2
google-auth>=2.30

# ── TikTok Live (OPCIONAL): chat y reproductor de directos de TikTok ──────────
# Librería NO oficial (como pytchat para YouTube). Solo lectura, sin login.
# betterproto va pineado en su beta porque TikTokLive 6.x lo exige y uv no
# instala prereleases sin un pin explícito. httpx se limita a <1 para que
# convivan pytchat y TikTokLive. Sin estas dependencias, la rama TikTok avisa
# y el resto de la app funciona igual.
betterproto==2.0.0b7
httpx>=0.25,<1
TikTokLive>=6.6,<7

# ── Reproductor de audio (OPCIONAL) ───────────────────────────────────────────
# Para el reproductor: yt-dlp extrae la URL del flujo de audio y libVLC
# (python-vlc) lo reproduce. python-vlc necesita VLC instalado en el equipo
# (videolan.org); el paquete distribuible lleva libVLC dentro, así que el amigo
# no instala nada. Si falta cualquiera, el reproductor muestra un aviso y el
# resto de la app funciona igual.
yt-dlp>=2025.1.1
python-vlc>=3.0.20
